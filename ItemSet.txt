import java.io.File;
import java.io.IOException;
import java.io.PrintWriter;
import java.util.ArrayList;
import java.util.Collections;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Scanner;
import java.util.Set;
import java.util.StringTokenizer;


public class mining {

	public static void main(String args[]) throws IOException{
		
		File file = new File("C:/Users/KRITHIVASAN CHANDRAN/Desktop/proj_1/transactionDB.txt");
		PrintWriter writer = new PrintWriter("C:/Users/KRITHIVASAN CHANDRAN/Desktop/proj_1/binaryTransactionDB.txt", "UTF-8");
		PrintWriter writer1 = new PrintWriter("C:/Users/KRITHIVASAN CHANDRAN/Desktop/proj_1/UniqueItemSet.txt", "UTF-8");
		File newFile = new File("C:/Users/KRITHIVASAN CHANDRAN/Desktop/proj_1/transactionDB.txt");
		Scanner newinput = new Scanner(newFile);
		newinput.useDelimiter(System.getProperty("line.separator")); 
		int counter = 0;
		int k = 8;
		int p=0;
		Scanner input = new Scanner(file);
		Set<String> s = new HashSet<>();
		StringBuffer buffer = new StringBuffer();
		Map<String,String> itemSetMap = new HashMap<>();
		List<String> transacDB = new ArrayList<String>();
		List<String> modifiedList = new ArrayList<String>();
	        
		while(input.hasNext()){
			    	String line = input.next();
			        line = line.replaceAll("\\s+",",");
			        
			        StringTokenizer tokens = new  StringTokenizer(line,",");
			        
			        while (tokens.hasMoreTokens()){
			           String temp = tokens.nextToken();
			           s.add(temp);
			           transacDB.add(temp);
			         }
		}
		input.close();
		
        List<String> itemSet = new ArrayList<>();
        Map<Integer,List<String>> subset = new HashMap<>();
        int i =0;
        
        for(String str:s){
        	itemSet.add(str);
        }
        
        Collections.sort(itemSet);
        
        //Generating key value pairs - Key - Binary Numbers - Value - UniqueItemSets
      
        while(p<itemSet.size()){
        	writer1.append(Integer.toBinaryString(p));
        	writer1.append("=");
        	writer1.append(itemSet.get(p));
        	writer1.append("\n");
        	itemSetMap.put(Integer.toBinaryString(p), itemSet.get(p));
			p++;
		}
        
        //Replacing the Contents of TransactionDB with Binary Numbers Equivalent in ITEMSETMAP.
        
        while(newinput.hasNext()){
    	String line = newinput.nextLine();
    	String amy  = line.replaceAll("\\s+",",");
    	StringTokenizer tokenizer = new  StringTokenizer(amy,",");
    	while(tokenizer.hasMoreTokens()){
    		String trans = tokenizer.nextToken(",");
    		if(itemSetMap.containsValue(trans)){
    			Set<String> keys = itemSetMap.keySet();
    			for(String key : keys){
    				String oldTransactionDB = itemSetMap.get(key);
    				if(trans.equalsIgnoreCase(oldTransactionDB)){
    					String replacedString = trans.replace(trans, key);
    			        synchronized (writer) {
	    					writer.append(replacedString);
	    					writer.append("  ");
    			        }
    				}
    			}
    		}
    	}	writer.append("\n");
        }

        s.clear();
   
        
        while((k)<itemSet.size()){
                
        		for(int j=i;j<k;j++){
                	
                	String candidateKey = itemSet.get(j);
                	buffer.append(candidateKey+",");
                }
                counter = counter + 1;
                buffer.append("---"+counter);
                writer.append(buffer.toString());
                buffer.delete(0, buffer.length());
                i = i+1;
                k = k+1;
        	 
        }
        writer.close();
        	}
        }
